Usage
<!-- Basic table -->
<table class="mdst-table">
  <thead>
    <tr>
      <th>Header</th>
      <th>Header</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>Cell</td>
      <td>Cell</td>
    </tr>
  </tbody>
</table>

<!-- With scroll wrapper -->
<div class="mdst-table-wrap">
  <table class="mdst-table">...</table>
</div>

<!-- Stack on mobile -->
<table class="mdst-table mdst-table--stack">...</table>

<!-- Combined variants -->
<table class="mdst-table mdst-table--bordered mdst-table--striped">...</table>
